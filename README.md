  <script>
    document.addEventListener('DOMContentLoaded', () => {
      const container = document.querySelector('.container');
      const editBtn = document.querySelector('.edit-btn');
      const saveBtn = document.querySelector('.save-btn');
      const cancelBtn = document.querySelector('.cancel-btn');
      const editableElements = document.querySelectorAll('.editable');

      editBtn.addEventListener('click', () => {
        container.classList.add('edit-mode');
        editBtn.parentElement.classList.add('hidden');
        saveBtn.parentElement.classList.remove('hidden');
        editableElements.forEach(element => {
          element.contentEditable = 'true';
          element.classList.add('edit-mode');
        });
      });

      saveBtn.addEventListener('click', () => {
        container.classList.remove('edit-mode');
        editBtn.parentElement.classList.remove('hidden');
        saveBtn.parentElement.classList.add('hidden');
        editableElements.forEach(element => {
          element.contentEditable = 'false';
          element.classList.remove('edit-mode');
        });
      });

      cancelBtn.addEventListener('click', () => {
        location.reload(); // Reloads the page to revert changes
      });
    });
  </script>
